\DoxyCodeLine{20   \textcolor{comment}{// Given a set of 3D points P, generate data structures for a pointerless}}
\DoxyCodeLine{21   \textcolor{comment}{// octree. Each cell stores its points, children, center location and width.}}
\DoxyCodeLine{22   \textcolor{comment}{// Our octree is not dense. We use the following rule: if the current cell}}
\DoxyCodeLine{23   \textcolor{comment}{// has any number of points, it will have all 8 children. A leaf cell will}}
\DoxyCodeLine{24   \textcolor{comment}{// have -\/1's as its list of child indices.}}
\DoxyCodeLine{25   \textcolor{comment}{//}}
\DoxyCodeLine{26   \textcolor{comment}{// We use a binary numbering of children. Treating the parent cell's center}}
\DoxyCodeLine{27   \textcolor{comment}{// as the origin, we number the octants in the following manner:}}
\DoxyCodeLine{28   \textcolor{comment}{// The first bit is 1 iff the octant's x coordinate is positive}}
\DoxyCodeLine{29   \textcolor{comment}{// The second bit is 1 iff the octant's y coordinate is positive}}
\DoxyCodeLine{30   \textcolor{comment}{// The third bit is 1 iff the octant's z coordinate is positive}}
\DoxyCodeLine{31   \textcolor{comment}{//}}
\DoxyCodeLine{32   \textcolor{comment}{// For example, the octant with negative x, positive y, positive z is:}}
\DoxyCodeLine{33   \textcolor{comment}{// 110 binary = 6 decimal}}
\DoxyCodeLine{34   \textcolor{comment}{//}}
\DoxyCodeLine{35   \textcolor{comment}{// Inputs:}}
\DoxyCodeLine{36   \textcolor{comment}{//   P  \#P by 3 list of point locations}}
\DoxyCodeLine{37   \textcolor{comment}{//}}
\DoxyCodeLine{38   \textcolor{comment}{// Outputs:}}
\DoxyCodeLine{39   \textcolor{comment}{//   point\_indices  a vector of vectors, where the ith entry is a vector of}}

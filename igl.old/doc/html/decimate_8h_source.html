15 // Assumes (V,F) is a manifold mesh (possibly with boundary) Collapses edges
16 // until desired number of faces is achieved. This uses default edge cost and
17 // merged vertex placement functions {edge length, edge midpoint}.
18 //
19 // Inputs:
20 // V #V by dim list of vertex positions
21 // F #F by 3 list of face indices into V.
22 // max_m desired number of output faces
23 // Outputs:
24 // U #U by dim list of output vertex posistions (can be same ref as V)
25 // G #G by 3 list of output face indices into U (can be same ref as G)
26 // J #G list of indices into F of birth face
27 // I #U list of indices into V of birth vertices
28 // Returns true if m was reached (otherwise #G > m)
29 IGL_INLINE bool decimate(
30 const Eigen::MatrixXd & V,
31 const Eigen::MatrixXi & F,
32 const size_t max_m,
33 Eigen::MatrixXd & U,
34 Eigen::MatrixXi & G,
35 Eigen::VectorXi & J,
36 Eigen::VectorXi & I);
37 // Inputs:
38 // V #V by dim list of vertex positions

40 // max_m desired number of output faces
41 // Outputs:
42 // U #U by dim list of output vertex posistions (can be same ref as V)
43 // G #G by 3 list of output face indices into U (can be same ref as G)

53 // Assumes a **closed** manifold mesh. See igl::connect_boundary_to_infinity
54 // and igl::decimate in decimate.cpp
55 // is handling meshes with boundary by connecting all boundary edges with
56 // dummy facets to infinity **and** modifying the stopping criteria.
57 //
58 // Inputs:
59 // cost_and_placement function computing cost of collapsing an edge and 3d
60 // position where it should be placed:
61 // cost_and_placement(V,F,E,EMAP,EF,EI,cost,placement);
62 // stopping_condition function returning whether to stop collapsing edges
63 // based on current state. Guaranteed to be called after _successfully_
64 // collapsing edge e removing edges (e,e1,e2) and faces (f1,f2):
65 // bool should_stop =
66 // stopping_condition(V,F,E,EMAP,EF,EI,Q,Qit,C,e,e1,e2,f1,f2);
67 IGL_INLINE bool decimate(
68 const Eigen::MatrixXd & V,
69 const Eigen::MatrixXi & F,
70 const decimate_cost_and_placement_callback & cost_and_placement,
71 const decimate_stopping_condition_callback & stopping_condition,
72 Eigen::MatrixXd & U,
73 Eigen::MatrixXi & G,
74 Eigen::VectorXi & J,
75 Eigen::VectorXi & I);
76 // Inputs:
77 // pre_collapse callback called with index of edge whose collapse is about
78 // to be attempted (see collapse_edge)
79 // post_collapse callback called with index of edge whose collapse was
80 // just attempted and a flag revealing whether this was successful (see
81 // collapse_edge)
82 IGL_INLINE bool decimate(
83 const Eigen::MatrixXd & V,
84 const Eigen::MatrixXi & F,
85 const decimate_cost_and_placement_callback & cost_and_placement,
86 const decimate_stopping_condition_callback & stopping_condition,
87 const decimate_pre_collapse_callback & pre_collapse,
88 const decimate_post_collapse_callback & post_collapse,
89 Eigen::MatrixXd & U,
90 Eigen::MatrixXi & G,
91 Eigen::VectorXi & J,
92 Eigen::VectorXi & I);
93 // Inputs:
94 // EMAP #F*3 list of indices into E, mapping each directed edge to unique
95 // unique edge in E
96 // EF #E by 2 list of edge flaps, EF(e,0)=f means e=(i-->j) is the edge of
97 // F(f,:) opposite the vth corner, where EI(e,0)=v. Similarly EF(e,1) "
98 // e=(j->i)
99 // EI #E by 2 list of edge flap corners (see above).

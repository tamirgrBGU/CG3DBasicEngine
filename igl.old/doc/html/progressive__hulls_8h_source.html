17 // Assumes (V,F) is a closed manifold mesh
18 // Collapses edges until desired number of faces is achieved but ensures
19 // that new vertices are placed outside all previous meshes as per
20 // "progressive hulls" in "Silhouette clipping" [Sander et al. 2000].
21 //
22 // Inputs:
23 // V #V by dim list of vertex positions
24 // F #F by 3 list of face indices into V.
25 // max_m desired number of output faces
26 // Outputs:
27 // U #U by dim list of output vertex posistions (can be same ref as V)
28 // G #G by 3 list of output face indices into U (can be same ref as G)
29 // J #G list of indices into F of birth faces
30 // Returns true if m was reached (otherwise #G > m)
31 IGL_INLINE bool progressive_hulls(
32 const Eigen::MatrixXd & V,
33 const Eigen::MatrixXi & F,
34 const size_t max_m,
35 Eigen::MatrixXd & U,
36 Eigen::MatrixXi & G,
37 Eigen::VectorXi & J);
38 }
39}

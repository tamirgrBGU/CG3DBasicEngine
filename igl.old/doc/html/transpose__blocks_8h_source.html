16 // Templates:
17 // T should be a eigen matrix primitive type like int or double
18 // Inputs:
19 // A m*k by n (dim: 1) or m by n*k (dim: 2) eigen Matrix of type T values
20 // k number of blocks
21 // dim dimension in which to transpose
22 // Output
23 // B n*k by m (dim: 1) or n by m*k (dim: 2) eigen Matrix of type T values,
24 // NOT allowed to be the same as A

27 // A = [

31 // 105 106 107 108
32 // 201 202 203 204
33 // 205 206 207 208];

35 // B -> [
36 // 1 5
37 // 2 6
38 // 3 7
39 // 4 8
40 // 101 105
41 // 102 106
42 // 103 107
43 // 104 108
44 // 201 205
45 // 202 206
46 // 203 207
47 // 204 208];
48 //
49 template <typename T>
50 IGL_INLINE void transpose_blocks(
51 const Eigen::Matrix<T,Eigen::Dynamic,Eigen::Dynamic> & A,
52 const size_t k,
53 const size_t dim,
54 Eigen::Matrix<T,Eigen::Dynamic,Eigen::Dynamic> & B);
55}

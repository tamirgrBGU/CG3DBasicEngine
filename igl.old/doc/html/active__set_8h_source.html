19 // Known Bugs: rows of [Aeq;Aieq] **must** be linearly independent. Should be
20 // using QR decomposition otherwise:
21 // https://v8doc.sas.com/sashtml/ormp/chap5/sect32.htm
22 //
23 // ACTIVE_SET Minimize quadratic energy
24 //
25 // 0.5*Z'*A*Z + Z'*B + C with constraints
26 //
27 // that Z(known) = Y, optionally also subject to the constraints Aeq*Z = Beq,
28 // and further optionally subject to the linear inequality constraints that
29 // Aieq*Z <= Bieq and constant inequality constraints lx <= x <= ux
30 //
31 // Inputs:
32 // A n by n matrix of quadratic coefficients
33 // B n by 1 column of linear coefficients
34 // known list of indices to known rows in Z
35 // Y list of fixed values corresponding to known rows in Z
36 // Aeq meq by n list of linear equality constraint coefficients
37 // Beq meq by 1 list of linear equality constraint constant values
38 // Aieq mieq by n list of linear inequality constraint coefficients
39 // Bieq mieq by 1 list of linear inequality constraint constant values
40 // lx n by 1 list of lower bounds [] implies -Inf
41 // ux n by 1 list of upper bounds [] implies Inf
42 // params struct of additional parameters (see below)
43 // Z if not empty, is taken to be an n by 1 list of initial guess values
44 // (see output)
45 // Outputs:
46 // Z n by 1 list of solution values
47 // Returns true on success, false on error
48 //
49 // Benchmark: For a harmonic solve on a mesh with 325K facets, matlab 2.2
50 // secs, igl/min_quad_with_fixed.h 7.1 secs
51 //
52 template <
53 typename AT,
54 typename DerivedB,
55 typename Derivedknown,
56 typename DerivedY,
57 typename AeqT,
58 typename DerivedBeq,
59 typename AieqT,
60 typename DerivedBieq,
61 typename Derivedlx,
62 typename Derivedux,
63 typename DerivedZ
64 >
65 IGL_INLINE igl::SolverStatus active_set(
66 const Eigen::SparseMatrix<AT>& A,
67 const Eigen::PlainObjectBase<DerivedB> & B,
68 const Eigen::PlainObjectBase<Derivedknown> & known,
69 const Eigen::PlainObjectBase<DerivedY> & Y,
70 const Eigen::SparseMatrix<AeqT>& Aeq,
71 const Eigen::PlainObjectBase<DerivedBeq> & Beq,
72 const Eigen::SparseMatrix<AieqT>& Aieq,
73 const Eigen::PlainObjectBase<DerivedBieq> & Bieq,
74 const Eigen::PlainObjectBase<Derivedlx> & lx,
75 const Eigen::PlainObjectBase<Derivedux> & ux,
76 const igl::active_set_params & params,
77 Eigen::PlainObjectBase<DerivedZ> & Z
78 );
79};

82struct igl::active_set_params
83{
84 // Input parameters for active_set:
85 // Auu_pd whether Auu is positive definite {false}
86 // max_iter Maximum number of iterations (0 = Infinity, {100})
87 // inactive_threshold Threshold on Lagrange multiplier values to determine
88 // whether to keep constraints active {EPS}
89 // constraint_threshold Threshold on whether constraints are violated (0
90 // is perfect) {EPS}
91 // solution_diff_threshold Threshold on the squared norm of the difference
92 // between two consecutive solutions {EPS}
93 bool Auu_pd;
94 int max_iter;
95 double inactive_threshold;
96 double constraint_threshold;
97 double solution_diff_threshold;
98 active_set_params():
99 Auu_pd(false),
100 max_iter(100),
101 inactive_threshold(igl::DOUBLE_EPS),
102 constraint_threshold(igl::DOUBLE_EPS),
103 solution_diff_threshold(igl::DOUBLE_EPS)
104 {};
105};

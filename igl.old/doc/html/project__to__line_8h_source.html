15 // PROJECT_TO_LINE project points onto vectors, that is find the parameter
16 // t for a point p such that proj_p = (y-x).*t, additionally compute the
17 // squared distance from p to the line of the vector, such that
18 // |p - proj_p|Â² = sqr_d
19 //
20 // [T,sqrD] = project_to_line(P,S,D)
21 //
22 // Inputs:
23 // P #P by dim list of points to be projected
24 // S size dim start position of line vector
25 // D size dim destination position of line vector
26 // Outputs:
27 // T #P by 1 list of parameters
28 // sqrD #P by 1 list of squared distances
29 //
30 //
31 template <
32 typename DerivedP,
33 typename DerivedS,
34 typename DerivedD,
35 typename Derivedt,
36 typename DerivedsqrD>
37 IGL_INLINE void project_to_line(
38 const Eigen::MatrixBase<DerivedP> & P,
39 const Eigen::MatrixBase<DerivedS> & S,
40 const Eigen::MatrixBase<DerivedD> & D,
41 Eigen::PlainObjectBase<Derivedt> & t,
42 Eigen::PlainObjectBase<DerivedsqrD> & sqrD);

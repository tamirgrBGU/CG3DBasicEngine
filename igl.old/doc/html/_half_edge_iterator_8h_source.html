20 // HalfEdgeIterator - Fake halfedge for fast and easy navigation
21 // on triangle meshes with vertex_triangle_adjacency and
22 // triangle_triangle adjacency
23 //
24 // Note: this is different to classical Half Edge data structure.
25 // Instead, it follows cell-tuple in [Brisson, 1989]
26 // "Representing geometric structures in d dimensions: topology and order."
27 // This class can achieve local navigation similar to half edge in OpenMesh
28 // But the logic behind each atom operation is different.
29 // So this should be more properly called TriangleTupleIterator.
30 //
31 // Each tuple contains information on (face, edge, vertex)
32 // and encoded by (face, edge \in {0,1,2}, bool reverse)
33 //
34 // Inputs:
35 // F #F by 3 list of "faces"
36 // FF #F by 3 list of triangle-triangle adjacency.
37 // FFi #F by 3 list of FF inverse. For FF and FFi, refer to
38 // "triangle_triangle_adjacency.h"
39 // Usages:
40 // FlipF/E/V changes solely one actual face/edge/vertex resp.
41 // NextFE iterates through one-ring of a vertex robustly.
42 //
43 template <
44 typename DerivedF,
45 typename DerivedFF,
46 typename DerivedFFi>
47 class HalfEdgeIterator
48 {
49 public:
50 // Init the HalfEdgeIterator by specifying Face,Edge Index and Orientation
51 IGL_INLINE HalfEdgeIterator(
52 const Eigen::MatrixBase<DerivedF>& _F,
53 const Eigen::MatrixBase<DerivedFF>& _FF,
54 const Eigen::MatrixBase<DerivedFFi>& _FFi,
55 int _fi,
56 int _ei,
57 bool _reverse = false
58 );
59
60 // Change Face
61 IGL_INLINE void flipF();
62
63 // Change Edge
64 IGL_INLINE void flipE();
65
66 // Change Vertex
67 IGL_INLINE void flipV();
68
69 IGL_INLINE bool isBorder();
70
84 IGL_INLINE bool NextFE();
85
86 // Get vertex index
87 IGL_INLINE int Vi();
88
89 // Get face index
90 IGL_INLINE int Fi();
91
92 // Get edge index
93 IGL_INLINE int Ei();
94
95 IGL_INLINE bool operator==(HalfEdgeIterator& p2);
96
97 private:
98 int fi;
99 int ei;
100 bool reverse;
101
102 // All the same type? This is likely to break.
103 const Eigen::MatrixBase<DerivedF> & F;
104 const Eigen::MatrixBase<DerivedFF> & FF;
105 const Eigen::MatrixBase<DerivedFFi> & FFi;
106 };

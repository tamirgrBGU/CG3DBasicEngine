17 // Concatenate k meshes into a single >=k connected component mesh with a
18 // single vertex list and face list. Similar to Maya's Combine operation.
19 //
20 // Inputs:
21 // VV k-long list of lists of mesh vertex positions
22 // FF k-long list of lists of mesh face indices so that FF[i] indexes
23 // VV[i]
24 // Outputs:
25 // V VV[0].rows()+...+VV[k-1].rows() by VV[0].cols() list of mesh
26 // vertex positions
27 // F FF[0].rows()+...+FF[k-1].rows() by FF[0].cols() list of mesh faces
28 // indices into V
29 // Vsizes k list so that Vsizes(i) is the #vertices in the ith input
30 // Fsizes k list so that Fsizes(i) is the #faces in the ith input
31 // Example:
32 // // Suppose you have mesh A (VA,FA) and mesh B (VB,FB)
33 // igl::combine<Eigen::MatrixXd,Eigen::MatrixXi>({VA,VB},{FA,FB},V,F);
34 //
35 //
36 template <
37 typename DerivedVV,
38 typename DerivedFF,
39 typename DerivedV,
40 typename DerivedF,
41 typename DerivedVsizes,
42 typename DerivedFsizes>
43 IGL_INLINE void combine(
44 const std::vector<DerivedVV> & VV,
45 const std::vector<DerivedFF> & FF,
46 Eigen::PlainObjectBase<DerivedV> & V,
47 Eigen::PlainObjectBase<DerivedF> & F,
48 Eigen::PlainObjectBase<DerivedVsizes> & Vsizes,
49 Eigen::PlainObjectBase<DerivedFsizes> & Fsizes);
50 template <
51 typename DerivedVV,
52 typename DerivedFF,
53 typename DerivedV,
54 typename DerivedF>
55 IGL_INLINE void combine(
56 const std::vector<DerivedVV> & VV,
57 const std::vector<DerivedFF> & FF,
58 Eigen::PlainObjectBase<DerivedV> & V,
59 Eigen::PlainObjectBase<DerivedF> & F);
60}

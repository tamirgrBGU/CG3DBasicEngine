71 // A \ B \ ... \ Z = A \ (B ∪ ... ∪ Z)
72 template <>
73 class BinaryWindingNumberOperations<MESH_BOOLEAN_TYPE_MINUS> {
74 public:
75 template<typename DerivedW>
76 typename DerivedW::Scalar operator()(
77 const Eigen::PlainObjectBase<DerivedW>& win_nums) const
78 {
79 assert(win_nums.size()>1);
80 // Union of objects 1 through n-1
81 bool union_rest = false;
82 for(int i = 1;i<win_nums.size();i++)
83 {
84 union_rest = union_rest || win_nums(i) > 0;
85 if(union_rest) break;
86 }
87 // Must be in object 0 and not in union of objects 1 through n-1
88 return win_nums(0) > 0 && !union_rest;
89 }
90 };
91
92 // A ∆ B ∆ ... ∆ Z (equivalent to set inside odd number of objects)
93 template <>
94 class BinaryWindingNumberOperations<MESH_BOOLEAN_TYPE_XOR> {
95 public:
96 template<typename DerivedW>
97 typename DerivedW::Scalar operator()(
98 const Eigen::PlainObjectBase<DerivedW>& win_nums) const
99 {
100 // If inside an odd number of objects
101 int count = 0;
102 for(int i = 0;i<win_nums.size();i++)
103 {
104 if(win_nums(i) > 0) count++;
105 }
106 return count % 2 == 1;
107 }
108 };

18 // Assumes (V,F) is a closed manifold mesh (except for previously collapsed
19 // faces which should be set to:
20 // [IGL_COLLAPSE_EDGE_NULL IGL_COLLAPSE_EDGE_NULL IGL_COLLAPSE_EDGE_NULL].
21 // Collapses exactly two faces and exactly 3 edges from E (e and one side of
22 // each face gets collapsed to the other). This is implemented in a way that
23 // it can be repeatedly called until satisfaction and then the garbage in F
24 // can be collected by removing NULL faces.
25 //
26 // Inputs:
27 // e index into E of edge to try to collapse. E(e,:) = [s d] or [d s] so
28 // that s<d, then d is collapsed to s.
30 // Inputs/Outputs:
31 // V #V by dim list of vertex positions, lesser index of E(e,:) will be set
32 // to midpoint of edge.
33 // F #F by 3 list of face indices into V.
34 // E #E by 2 list of edge indices into V.
35 // EMAP #F*3 list of indices into E, mapping each directed edge to unique
36 // unique edge in E
37 // EF #E by 2 list of edge flaps, EF(e,0)=f means e=(i-->j) is the edge of
38 // F(f,:) opposite the vth corner, where EI(e,0)=v. Similarly EF(e,1) "
39 // e=(j->i)
40 // EI #E by 2 list of edge flap corners (see above).
41 // e1 index into E of edge collpased on left
42 // e2 index into E of edge collpased on right
43 // f1 index into F of face collpased on left
44 // f2 index into F of face collpased on right
45 // Returns true if edge was collapsed

87 // Collapse least-cost edge from a priority queue and update queue
88 //
89 // Inputs/Outputs:
90 // cost_and_placement function computing cost of collapsing an edge and 3d
91 // position where it should be placed:
92 // cost_and_placement(V,F,E,EMAP,EF,EI,cost,placement);
93 // **If the edges is collapsed** then this function will be called on all
94 // edges of all faces previously incident on the endpoints of the
95 // collapsed edge.
96 // Q queue containing pairs of costs and edge indices and insertion "time"
97 // EQ #E list of "time" of last time pushed into Q
98 // C #E by dim list of stored placements
99 IGL_INLINE bool collapse_edge(
100 const decimate_cost_and_placement_callback & cost_and_placement,
101 Eigen::MatrixXd & V,
102 Eigen::MatrixXi & F,
103 Eigen::MatrixXi & E,

105 Eigen::MatrixXi & EF,
106 Eigen::MatrixXi & EI,
107 igl::min_heap< std::tuple<double,int,int> > & Q,
108 Eigen::VectorXi & EQ,
109 Eigen::MatrixXd & C);
110 // Inputs:
111 // pre_collapse callback called with index of edge whose collapse is about
112 // to be attempted. This function should return whether to **proceed**
113 // with the collapse: returning true means "yes, try to collapse",
114 // returning false means "No, consider this edge 'uncollapsable', behave
115 // as if collapse_edge(e) returned false.
116 // post_collapse callback called with index of edge whose collapse was

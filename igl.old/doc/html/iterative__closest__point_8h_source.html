16 // Solve for the rigid transformation that places mesh X onto mesh Y using the
17 // iterative closest point method. In particular, optimize:
18 //
19 // min ∫_X inf ‖x*R+t - y‖² dx
20 // R∈SO(3) y∈Y
21 // t∈R³
22 //
23 // Typically optimization strategies include using Gauss Newton
24 // ("point-to-plane" linearization) and stochastic descent (sparse random
25 // sampling each iteration).
26 //
27 // Inputs:
28 // VX #VX by 3 list of mesh X vertices
29 // FX #FX by 3 list of mesh X triangle indices into rows of VX
30 // VY #VY by 3 list of mesh Y vertices
31 // FY #FY by 3 list of mesh Y triangle indices into rows of VY
32 // num_samples number of random samples to use (larger --> more accurate,
33 // but also more suceptible to sticking to local minimum)
34 // Outputs:
35 // R 3x3 rotation matrix so that (VX*R+t,FX) ~~ (VY,FY)
36 // t 1x3 translation row vector
37 template <
38 typename DerivedVX,
39 typename DerivedFX,
40 typename DerivedVY,
41 typename DerivedFY,
42 typename DerivedR,
43 typename Derivedt
44 >
45 IGL_INLINE void iterative_closest_point(
46 const Eigen::MatrixBase<DerivedVX> & VX,
47 const Eigen::MatrixBase<DerivedFX> & FX,
48 const Eigen::MatrixBase<DerivedVY> & VY,
49 const Eigen::MatrixBase<DerivedFY> & FY,
50 const int num_samples,
51 const int max_iters,
52 Eigen::PlainObjectBase<DerivedR> & R,
53 Eigen::PlainObjectBase<Derivedt> & t);
54 // Inputs:
55 // Ytree precomputed AABB tree for accelerating closest point queries
56 // NY #FY by 3 list of precomputed unit face normals
57 template <
58 typename DerivedVX,
59 typename DerivedFX,
60 typename DerivedVY,
61 typename DerivedFY,
62 typename DerivedNY,
63 typename DerivedR,
64 typename Derivedt
65 >
66 IGL_INLINE void iterative_closest_point(
67 const Eigen::MatrixBase<DerivedVX> & VX,
68 const Eigen::MatrixBase<DerivedFX> & FX,
69 const Eigen::MatrixBase<DerivedVY> & VY,
70 const Eigen::MatrixBase<DerivedFY> & FY,
71 const igl::AABB<DerivedVY,3> & Ytree,
72 const Eigen::MatrixBase<DerivedNY> & NY,
73 const int num_samples,
74 const int max_iters,
75 Eigen::PlainObjectBase<DerivedR> & R,
76 Eigen::PlainObjectBase<Derivedt> & t);
77}

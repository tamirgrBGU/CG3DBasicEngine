16 // Compute boundary conditions for automatic weights computation. This
17 // function expects that the given mesh (V,Ele) has sufficient samples
18 // (vertices) exactly at point handle locations and exactly along bone and
19 // cage edges.
20 //
21 // Inputs:
22 // V #V by dim list of domain vertices
23 // Ele #Ele by simplex-size list of simplex indices
24 // C #C by dim list of handle positions
25 // P #P by 1 list of point handle indices into C
26 // BE #BE by 2 list of bone edge indices into C
27 // CE #CE by 2 list of cage edge indices into *P*
28 // Outputs:
29 // b #b list of boundary indices (indices into V of vertices which have
30 // known, fixed values)
31 // bc #b by #weights list of known/fixed values for boundary vertices
32 // (notice the #b != #weights in general because #b will include all the
33 // intermediary samples along each bone, etc.. The ordering of the
34 // weights corresponds to [P;BE]
35 // Returns false if boundary conditions are suspicious:
36 // P and BE are empty
37 // bc is empty
38 // some column of bc doesn't have a 0 (assuming bc has >1 columns)
39 // some column of bc doesn't have a 1 (assuming bc has >1 columns)
40 IGL_INLINE bool boundary_conditions(
41 const Eigen::MatrixXd & V,
42 const Eigen::MatrixXi & Ele,
43 const Eigen::MatrixXd & C,
44 const Eigen::VectorXi & P,
45 const Eigen::MatrixXi & BE,
46 const Eigen::MatrixXi & CE,
47 Eigen::VectorXi & b,
48 Eigen::MatrixXd & bc);
49}

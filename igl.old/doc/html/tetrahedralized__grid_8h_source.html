20 // Construct vertices of a regular grid, suitable for input to
21 // `igl::marching_tets`
22 //
23 // Inputs:
24 // nx number of grid vertices in x direction
25 // ny number of grid vertices in y direction
26 // nz number of grid vertices in z direction
27 // type type of tetrahedralization of cube to use
28 // Outputs:
29 // GV nx*ny*nz by 3 list of grid vertex positions
30 // GT (nx-1)*(ny-1)*(nz-1)*k by 4 list of tetrahedron indices into rows of
31 // V, where k is the number of tets per cube (dependent on type)
32 //
33 // See also: triangulated_grid, quad_grid
34 template <
35 typename DerivedGV,
36 typename DerivedGT>
37 IGL_INLINE void tetrahedralized_grid(
38 const int nx,
39 const int ny,
40 const int nz,
41 const TetrahedralizedGripType type,
42 Eigen::PlainObjectBase<DerivedGV> & GV,
43 Eigen::PlainObjectBase<DerivedGT> & GT);
44 //
45 // Inputs:
46 // GV nx*ny*nz by 3 list of grid vertex positions
47 // side 3-long list {nx,ny,nz} see above
48 // type type of tetrahedralization of cube to use
49 // Outputs:
50 // GT (nx-1)*(ny-1)*(nz-1)*k by 4 list of tetrahedron indices into rows of
51 // V, where k is the number of tets per cube (dependent on type)
52 //
53 template <
54 typename DerivedGV,
55 typename Derivedside,
56 typename DerivedGT>
57 IGL_INLINE void tetrahedralized_grid(
58 const Eigen::MatrixBase<DerivedGV> & GV,
59 const Eigen::MatrixBase<Derivedside> & side,
60 const TetrahedralizedGripType type,
61 Eigen::PlainObjectBase<DerivedGT> & GT);
62}

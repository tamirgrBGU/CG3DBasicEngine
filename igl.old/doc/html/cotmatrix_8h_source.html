24 // Constructs the cotangent stiffness matrix (discrete laplacian) for a given
25 // mesh (V,F).
26 //
27 // Templates:
28 // DerivedV derived type of eigen matrix for V (e.g. derived from
29 // MatrixXd)
30 // DerivedF derived type of eigen matrix for F (e.g. derived from
31 // MatrixXi)
32 // Scalar scalar type for eigen sparse matrix (e.g. double)
33 // Inputs:
34 // V #V by dim list of mesh vertex positions
35 // F #F by simplex_size list of mesh elements (triangles or tetrahedra)
36 // Outputs:
37 // L #V by #V cotangent matrix, each row i corresponding to V(i,:)
38 //
39 // See also: adjacency_matrix
40 //
41 // Note: This Laplacian uses the convention that diagonal entries are
42 // **minus** the sum of off-diagonal entries. The diagonal entries are
43 // therefore in general negative and the matrix is **negative** semi-definite
44 // (immediately, -L is **positive** semi-definite)
45 //
46 template <typename DerivedV, typename DerivedF, typename Scalar>
47 IGL_INLINE void cotmatrix(
48 const Eigen::MatrixBase<DerivedV> & V,
49 const Eigen::MatrixBase<DerivedF> & F,
50 Eigen::SparseMatrix<Scalar>& L);
51 // Cotangent Laplacian (and mass matrix) for polygon meshes according to
52 // "Polygon Laplacian Made Simple" [Bunge et al. 2020]
53 //
54 // Inputs:
55 // V #V by 3 list of mesh vertex positions
56 // I #I vectorized list of polygon corner indices into rows of some matrix V
57 // C #polygons+1 list of cumulative polygon sizes so that C(i+1)-C(i) = size of
58 // the ith polygon, and so I(C(i)) through I(C(i+1)-1) are the indices of
59 // the ith polygon
60 // Outputs:
61 // L #V by #V polygon Laplacian made simple matrix
62 // M #V by #V mass matrix
63 // P #V+#polygons by #V prolongation operator
64 template <
65 typename DerivedV,
66 typename DerivedI,
67 typename DerivedC,
68 typename Scalar>
69 IGL_INLINE void cotmatrix(
70 const Eigen::MatrixBase<DerivedV> & V,
71 const Eigen::MatrixBase<DerivedI> & I,
72 const Eigen::MatrixBase<DerivedC> & C,
73 Eigen::SparseMatrix<Scalar>& L,
74 Eigen::SparseMatrix<Scalar>& M,
75 Eigen::SparseMatrix<Scalar>& P);
76}

67 // Compute necessary information to start using an ARAP deformation
68 //
69 // Inputs:
70 // V #V by dim list of mesh positions
71 // F #F by simplex-size list of triangle|tet indices into V
72 // dim dimension being used at solve time. For deformation usually dim =
73 // V.cols(), for surface parameterization V.cols() = 3 and dim = 2
74 // b #b list of "boundary" fixed vertex indices into V
75 // Outputs:
76 // data struct containing necessary precomputation
77 template <
78 typename DerivedV,
79 typename DerivedF,
80 typename Derivedb>
81 IGL_INLINE bool arap_precomputation(
82 const Eigen::MatrixBase<DerivedV> & V,
83 const Eigen::MatrixBase<DerivedF> & F,
84 const int dim,
85 const Eigen::MatrixBase<Derivedb> & b,
86 ARAPData & data);
87 // Inputs:
88 // bc #b by dim list of boundary conditions
89 // data struct containing necessary precomputation and parameters
90 // U #V by dim initial guess
91 //
92 // NOTE: While the libigl guidelines require outputs to be of type
93 // PlainObjectBase so that the user does not need to worry about allocating
94 // memory for the output, in this case, the user is required to give an initial
95 // guess and hence fix the size of the problem domain.
96 // Taking a reference to MatrixBase in this case thus allows the user to provide e.g.
97 // a map to the position data, allowing seamless interoperability with user-defined
98 // datastructures without requiring a copy.
99 template <
100 typename Derivedbc,
101 typename DerivedU>
102 IGL_INLINE bool arap_solve(
103 const Eigen::MatrixBase<Derivedbc> & bc,
104 ARAPData & data,
105 Eigen::MatrixBase<DerivedU> & U);
106};

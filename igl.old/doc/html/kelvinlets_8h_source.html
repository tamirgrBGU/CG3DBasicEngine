26 KelvinletParams(const Scalar& epsilon,
27 const int falloff,
28 const BrushType& type)
29 : epsilon(epsilon)
30 , scale(falloff)
31 , brushType(type)

33 static constexpr std::array<Scalar, 3> brush_scaling_params{ 1.0f,

36 for (int i = 0; i < 3; i++) {
37 ep[i] = epsilon * brush_scaling_params[i];

39 w[0] = 1;
40 w[1] = -((ep[2] * ep[2] - ep[0] * ep[0]) / (ep[2] * ep[2] - ep[1] * ep[1]));
41 w[2] = (ep[1] * ep[1] - ep[0] * ep[0]) / (ep[2] * ep[2] - ep[1] * ep[1]);
42 }
43};

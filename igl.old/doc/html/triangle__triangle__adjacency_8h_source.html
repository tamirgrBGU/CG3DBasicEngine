16 // Constructs the triangle-triangle adjacency matrix for a given
17 // mesh (V,F).
18 //
19 // Inputs:
20 // F #F by simplex_size list of mesh faces (must be triangles)
21 // Outputs:
22 // TT #F by #3 adjacent matrix, the element i,j is the id of the triangle
23 // adjacent to the j edge of triangle i
24 // TTi #F by #3 adjacent matrix, the element i,j is the id of edge of the
25 // triangle TT(i,j) that is adjacent with triangle i
26 //
27 // NOTE: the first edge of a triangle is [0,1] the second [1,2] and the third
28 // [2,3]. this convention is DIFFERENT from
29 // cotmatrix_entries.h/edge_lengths.h/etc. To fix this you could use:
30 // // Fix mis-match convention
31 // {
32 // Eigen::PermutationMatrix<3,3> perm(3);
33 // perm.indices() = Eigen::Vector3i(1,2,0);
34 // TT = (TT*perm).eval();
35 // TTi = (TTi*perm).eval();
36 // for(int i=0;i<TTi.rows();i++)
37 // for(int j=0;j<TTi.cols();j++)
38 // TTi(i,j)=TTi(i,j)==-1?-1:(TTi(i,j)+3-1)%3;
39 // }
40 template <typename DerivedF, typename DerivedTT, typename DerivedTTi>
41 IGL_INLINE void triangle_triangle_adjacency(
42 const Eigen::MatrixBase<DerivedF>& F,
43 Eigen::PlainObjectBase<DerivedTT>& TT,
44 Eigen::PlainObjectBase<DerivedTTi>& TTi);
45 template <typename DerivedF, typename DerivedTT>
46 IGL_INLINE void triangle_triangle_adjacency(
47 const Eigen::MatrixBase<DerivedF>& F,
48 Eigen::PlainObjectBase<DerivedTT>& TT);

66 // Adjacency list version, which works with non-manifold meshes
67 //
68 // Inputs:
69 // F #F by 3 list of triangle indices
70 // Outputs:
71 // TT #F by 3 list of lists so that TT[i][c] --> {j,k,...} means that
72 // faces j and k etc. are edge-neighbors of face i on face i's edge
73 // opposite corner c
74 // TTj #F list of lists so that TTj[i][c] --> {j,k,...} means that face
75 // TT[i][c][0] is an edge-neighbor of face i incident on the edge of face
76 // TT[i][c][0] opposite corner j, and TT[i][c][1] " corner k, etc.
77 template <
78 typename DerivedF,
79 typename TTIndex,
80 typename TTiIndex>

82 const Eigen::MatrixBase<DerivedF> & F,

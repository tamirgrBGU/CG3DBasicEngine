33 // Solve a convex quadratic optimization problem with linear and constant
34 // bounds, that is:
35 //
36 // Minimize: ½ * xT * Q⁰ * x + cT * x + cf
37 //
38 // Subject to: lc ≤ Ax ≤ uc
39 // lx ≤ x ≤ ux
40 //
41 // where we are trying to find the optimal vector of values x.
42 //
43 // Note: Q⁰ must be symmetric and the ½ is a convention of MOSEK
44 //
45 // Note: Because of how MOSEK accepts different parts of the system, Q
46 // should be stored in IJV (aka Coordinate) format and should only include
47 // entries in the lower triangle. A should be stored in Column compressed
48 // (aka Harwell Boeing) format. As described:
49 // http://netlib.org/linalg/html_templates/node92.html
50 // or
51 // http://en.wikipedia.org/wiki/Sparse_matrix
52 // #Compressed_sparse_column_.28CSC_or_CCS.29
53 //
54 //
55 // Templates:
56 // Index type for index variables
57 // Scalar type for floating point variables (gets cast to double?)
58 // Input:
59 // n number of variables, i.e. size of x
60 // Qi vector of qnnz row indices of non-zeros in LOWER TRIANGLE ONLY of
61 // Q⁰
62 // Qj vector of qnnz column indices of non-zeros in LOWER TRIANGLE ONLY
63 // of Q⁰
64 // Qv vector of qnnz values of non-zeros in LOWER TRIANGLE ONLY of Q⁰,
65 // such that:
66 //
67 // Q⁰(Qi[k],Qj[k]) = Qv[k] for k ∈ [0,Qnnz-1], where Qnnz is the
68 //
69 // number of non-zeros in Q⁰
70 // c (optional) vector of n values of c, transpose of coefficient row
71 // vector of linear terms, EMPTY means c == 0
72 // cf (ignored) value of constant term in objective, 0 means cf == 0, so
73 // optional only in the sense that it is mandatory
74 // m number of constraints, therefore also number of rows in linear
75 // constraint coefficient matrix A, and in linear constraint bound
76 // vectors lc and uc
77 // Av vector of non-zero values of A, in column compressed order
78 // Ari vector of row indices corresponding to non-zero values of A,
79 // Acp vector of indices into Ari and Av of the first entry for each
80 // column of A, size(Acp) = (# columns of A) + 1 = n + 1
81 // lc vector of m linear constraint lower bounds
82 // uc vector of m linear constraint upper bounds
83 // lx vector of n constant lower bounds
84 // ux vector of n constant upper bounds
85 // Output:
86 // x vector of size n to hold output of optimization
87 // Return:
88 // true only if optimization was successful with no errors
89 //
90 // Note: All indices are 0-based
91 //
92 template <typename Index, typename Scalar>
93 IGL_INLINE bool mosek_quadprog(
94 const Index n,
95 /* mosek won't allow this to be const*/ std::vector<Index> & Qi,
96 /* mosek won't allow this to be const*/ std::vector<Index> & Qj,
97 /* mosek won't allow this to be const*/ std::vector<Scalar> & Qv,
98 const std::vector<Scalar> & c,
99 const Scalar cf,
100 const Index m,
101 /* mosek won't allow this to be const*/ std::vector<Scalar> & Av,
102 /* mosek won't allow this to be const*/ std::vector<Index> & Ari,
103 const std::vector<Index> & Acp,
104 const std::vector<Scalar> & lc,
105 const std::vector<Scalar> & uc,
106 const std::vector<Scalar> & lx,
107 const std::vector<Scalar> & ux,
108 MosekData & mosek_data,
109 std::vector<Scalar> & x);
110 // Wrapper with Eigen elements
111 //
112 // Inputs:
113 // Q n by n square quadratic coefficients matrix **only lower triangle
114 // is used**.
115 // c n-long vector of linear coefficients
116 // cf constant coefficient
117 // A m by n square linear coefficienst matrix of inequality constraints
118 // lc m-long vector of lower bounds for linear inequality constraints
119 // uc m-long vector of upper bounds for linear inequality constraints
120 // lx n-long vector of lower bounds
121 // ux n-long vector of upper bounds
122 // mosek_data parameters struct
123 // Outputs:
124 // x n-long solution vector
125 // Returns true only if optimization finishes without error
126 //
127 IGL_INLINE bool mosek_quadprog(
128 const Eigen::SparseMatrix<double> & Q,
129 const Eigen::VectorXd & c,
130 const double cf,
131 const Eigen::SparseMatrix<double> & A,
132 const Eigen::VectorXd & lc,
133 const Eigen::VectorXd & uc,
134 const Eigen::VectorXd & lx,
135 const Eigen::VectorXd & ux,
136 MosekData & mosek_data,
137 Eigen::VectorXd & x);
138 }

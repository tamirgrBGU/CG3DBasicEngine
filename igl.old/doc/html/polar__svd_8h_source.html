15 // Computes the polar decomposition (R,T) of a matrix A using SVD singular
16 // value decomposition
17 //

24 // S 3 by 1 singular values
25 // V 3 by 3 right-singular vectors
26 //
27 //

33 typename DerivedS,
34 typename DerivedV>

37 Eigen::PlainObjectBase<DerivedR> & R,
38 Eigen::PlainObjectBase<DerivedT> & T,
